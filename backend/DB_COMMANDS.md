# أوامر إدارة قاعدة البيانات - RetailERP System
# ================================================

## 1. إنشاء ملفات الهجرة (Migrations)
# ------------------------------------
# هذا الأمر ينشئ ملفات الهجرة بناءً على التغييرات في models.py
python manage.py makemigrations core

## 2. تطبيق الهجرة وإنشاء قاعدة البيانات
# -----------------------------------------
# هذا الأمر يطبق جميع ملفات الهجرة وينشئ جداول قاعدة البيانات
python manage.py migrate

## 3. إنشاء مستخدم مسؤول (Superuser)
# ------------------------------------
# لإنشاء مدير نظام للدخول إلى لوحة التحكم
python manage.py createsuperuser

## 4. تشغيل خادم التطوير
# -----------------------
# لتشغيل сервер Django للتطوير
python manage.py runserver

## 5. التحقق من صحة النماذج
# --------------------------
# للتحقق من أن النماذج صحيحة ولا توجد أخطاء
python manage.py check

## 6. عرض تفاصيل قاعدة البيانات
# -------------------------------
# للدخول إلى Shell التفاعلي
python manage.py shell
